write_files:
  - path: /etc/rsyslog.d/39-logstash.conf
    content: |
      $ModLoad omrelp
      *.* :omrelp:10.79.180.239:2514
  - path: /etc/init/overview-worker.conf
    content: |
      description "Overview worker"
      author "Overview Project"
      setuid overview
      setgid overview
      chdir /opt/overview
      console output
      respawn
      exec /opt/overview/run-worker.sh worker JobHandler local1.notice 2500m
  - path: /etc/init/overview-documentset-worker.conf
    content: |
      description "Overview document-set worker"
      author "Overview Project"
      setuid overview
      setgid overview
      chdir /opt/overview
      console output
      respawn
      exec /opt/overview/run-worker.sh documentset-worker org.overviewproject.DocumentSetWorker local2.notice 700m
  - path: /opt/overview/run-worker.sh
    permissions: '0755'
    content: |
      #!/bin/bash

      SERVER=$1 # "worker" or "documentset-worker"
      MAIN_CLASS=$2 # "JobHandler" or "org.overviewproject.DocumentSetWorker"
      SYSLOG_OUTPUT=$3 # "local1.notice" or "local2.notice"
      MAX_MEMORY=$4 # "2500m" or "700m"

      aws s3 cp s3://overview-staging-secrets/worker-env.sh /tmp/$SERVER-env.sh
      source /tmp/$SERVER-env.sh

      function tag_to_ip() {
        aws ec2 describe-instances \
          --filter "Name=tag:Name,Values=$1" \
          --query "Reservations[*].Instances[0].PrivateIpAddress" \
          --output text
      }

      DATABASE_SERVER_NAME=$(tag_to_ip $DATABASE_SERVER_NAME_TAG)
      ES_HOSTS="$(tag_to_ip $ES_HOSTS_TAG):$ES_HOSTS_PORT"
      MESSAGE_BROKER_URI="tcp://$(tag_to_ip $MESSAGE_BROKER_URI_TAG):$MESSAGE_BROKER_PORT"

      aws cp s3://overview-releases/latest/archive.zip /tmp/$SERVER-archive.zip
      FILENAMES=$(unzip -p /tmp/$SERVER-archive.zip archive/$SERVER/classpath.txt | sed -e 's/^/archive\/lib\//')
      rm -r /opt/overview/$SERVER
      unzip -d /opt/overview/$SERVER -j $FILENAMES

      exec /usr/bin/java \
        -cp "/opt/overview/$SERVER/*" \
        -Dfile.encoding=UTF8 \
        -Dconfig.resource=production.conf \
        -Dlogback.configurationFile=workerprodlog.xml \
        -Duser.timezone=UTC \
        -Xmx$MAX_MEMORY \
        $MAIN_CLASS \
        2>&1 | logger -p "$SYSLOG_OUTPUT"
users:
  - default
  - name: overview
    home: /opt/overview
runcmd:
  - echo "127.0.1.1 $(hostname)" | sudo tee -a /etc/hosts
  - /etc/init.d/rsyslog stop
  - /etc/init.d/rsyslog start
  - start worker
  - start documentset-worker
